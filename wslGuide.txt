Server Setups:
wget https://repo.anaconda.com/miniconda/Miniconda3-latest-Linux-x86_64.sh
chmod +x Miniconda3-latest-Linux-x86_64.sh
./Miniconda3-latest-Linux-x86_64.sh

sudo apt update && sudo apt install build-essential
sudo apt-get install apt-transport-https ca-certificates curl gnupg lsb-release

curl -fsSL https://download.docker.com/linux/ubuntu/gpg | sudo gpg --dearmor -o /usr/share/keyrings/docker-archive-keyring.gpg

echo \
  "deb [arch=$(dpkg --print-architecture) signed-by=/usr/share/keyrings/docker-archive-keyring.gpg] https://download.docker.com/linux/ubuntu \
  $(lsb_release -cs) stable" | sudo tee /etc/apt/sources.list.d/docker.list > /dev/null

sudo apt-get update
sudo apt-get install docker-ce docker-ce-cli containerd.io
sudo usermod -aG docker $USER
sudo service docker start

sudo apt-get install git
curl -o- https://raw.githubusercontent.com/nvm-sh/nvm/v0.39.1/install.sh | bash
source ~/.bashrc
nvm install 22.10
git clone https://github.com/open-webui/open-webui.git

Terminal 1:
cd open-webui
cp -RPp .env.example .env
npm install --force
touch .env.local
--open that file and add       VITE_API_BASE_URL=http://localhost:8080
npm run dev



Terminal 2:
cd backend
conda create --name open-webui python=3.11
conda activate open-webui
pip install -r requirements.txt -U
sh dev.sh




Terminal 3:
conda create -n vllm_env python=3.11
conda activate vllm_env
pip install "vllm>=0.4.0" openai

docker run -d -p 9099:9099 \
--add-host=host.docker.internal:host-gateway \
-v pipelines:/app/pipelines \
--name pipelines \
--restart always ghcr.io/open-webui/pipelines:main

python3 -m vllm.entrypoints.openai.api_server --model "Qwen/Qwen3-4B-Thinking-2507" --max-model-len 16384
